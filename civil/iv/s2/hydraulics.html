Equations
$$\displaylines{ V=\frac{Q}{A}\quad\quad A=T\,y\\\\ A=T\,D\quad\quad Fr=\frac{V}{\sqrt{g\,D}}\\\\ \frac{dy}{dx}=\frac{S_0-S_f}{1-Fr^2}\\ \frac{dy}{dx}=+\Rightarrow\textsf{backwater curve}\\ \frac{dy}{dx}=-\Rightarrow\textsf{drawdown curve} }$$

Flow Profiles
$$\displaylines{ \textsf{Twelve of flow profiles are gradually varied flow.}\\ \textsf{M for mild (subcritical)} = M_1\,,\,M_2\,,\,M_3\\ \textsf{C for critical} = C_1\,,\,C_2\,,\,C_3\\ \textsf{S for steep (supercritical)} = S_1\,,\,S_2\,,\,S_3\\ \textsf{H for horizontal} = H_2\,,\,H_3\\ \textsf{A for adverse} = A_2\,,\,A_3\\ C_2 = \textsf{uniform flow}\\\\ \textsf{Zone I and III}\Rightarrow\textsf{backwater curve}\\ \textsf{Zone II}\Rightarrow\textsf{drawdown curve}\\\\ y=\textsf{actual depth}\\ y_n=\textsf{normal depth}\\ y_c=\textsf{critical depth}\\ NDL=\textsf{normal depth line}\\ CDL=\textsf{critical depth line}\\ y_n\gt y_c\Rightarrow\textsf{mild slope}\\ y_c\gt y_n\Rightarrow\textsf{steep slope}\\ y_n=y_c\Rightarrow NDL=CDL\Rightarrow\textsf{critical slope}\\ y_n=\infty\Rightarrow\textsf{horizontal, adverse slope} }$$

Location of Jump
$$\displaylines{ \textsf{For wide channel}\,,\,R\approx y\\\\ Q=\frac{1}{n}\,AR^{\frac{2}{3}}\,S^{\frac{1}{2}}\;\quad Q=AV=AC\sqrt{RS_0}\\\\q\;(m^3/s\;\textsf{per meter})=\frac{Q}{b}\quad\quad V=\frac{q}{y}\\\\ E=y+\frac{V^2}{2\,g}\\\\ y_c=\sqrt[3]{\frac{\alpha\,Q^2}{g\,b^2}}=\sqrt[3]{\frac{q^2}{g}}\quad\quad \alpha =1\\\\ y_n\lt y_c\Rightarrow\textsf{Steep slope}\\ y_n\gt y_c\Rightarrow\textsf{Mild slope}\\\\ \textsf{Assume jump will occur in}\\\textsf{steep slope}\;(y_2\lt y_{n_2})\\ y_1=y_{n_1}\quad\quad Fr^2=\frac{q^2}{g\,y_{n_1}^3}\\\\ \textsf{Depth after jump (sequent depth)}\,,\\ y_2=\frac{y_1}{2}\,\left[\,\sqrt{1+8\,Fr^2}-1\,\right]\\\\ \textsf{If}\;y_2\gt y_{n_2}\,,\;\textsf{jump will occur in mild slope.}\\ y_2=y_{n_2}\quad\quad Fr^2=\frac{q^2}{g\,y_{n_2}^3}\\\\ \textsf{Initial depth}\,,\,y_1=\frac{y_2}{2}\,\left[\,\sqrt{1+8\,Fr^2}-1\,\right]\\\\ E=y+\frac{\alpha\,q^2}{2\,g\,y^2}\quad\quad S_f=\frac{n^2\,q^2}{y^{\frac{10}{3}}}\\\\ \bar{S_f}=\frac{S_{f_1}+S_{f_2}}{2}\quad\quad \Delta x=\frac{E_2-E_1}{S_0-\bar{S_f}}\\\\ \Delta x=\frac{\Delta E}{S_0-\bar{S_f}} }$$

Critical Depth Control Point (C.D.C.P)
$$\displaylines{ \textsf{At critical depth control point (C.D.C.P)}\,,\\y=y_c\\\\ \frac{dy}{dx}=\frac{S_0-S_f}{1-Fr^2}=0\\\\ S_0-S_f=0\\\\ y_c=\sqrt[3]{\frac{Q^2}{g\,b^2}}\\\\ \textsf{At C.D.C.P}\,,\,\frac{dy}{dx}=0\\\\ \frac{dy}{dx}=\frac{S_0-S_f+\left(\frac{\alpha\,Q^2\,y}{g\,A^3}\right)\,\frac{db}{dx}}{\left(1-\frac{\alpha\,Q^2\,b}{g\,A^3}\right)}=0\\\\ S_f=\frac{n^2\,V^2}{R^{\frac{4}{3}}}\quad\textsf{(SI System)}\\\\ S_f=\frac{n^2\,V^2}{2.22\,R^{\frac{4}{3}}}\quad\textsf{(FPS System)} }$$
